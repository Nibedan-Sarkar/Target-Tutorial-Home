// ২. Firebase ইনিশিয়ালাইজ করুন firebase.initializeApp(firebaseConfig); const auth = firebase.auth(); // ৩. Signup ফাংশন document.querySelector('#signup-form form').addEventListener('submit', function(e) { e.preventDefault(); const email = this.querySelector('input[type="email"]').value; const password = this.querySelector('input[type="password"]').value; auth.createUserWithEmailAndPassword(email, password) .then(user => { alert("Signup successful!"); this.reset(); }) .catch(error => { alert(error.message); }); }); // ৪. Login ফাংশন document.querySelector('#login-form form').addEventListener('submit', function(e) { e.preventDefault(); const email = this.querySelector('input[type="email"]').value; const password = this.querySelector('input[type="password"]').value; auth.signInWithEmailAndPassword(email, password) .then(user => { alert("Login successful!"); this.reset(); // এখান থেকে Student Dashboard বা অন্য পেইজে রিডাইরেক্ট করতে পারেন }) .catch(error => { alert(error.message); }); });
